## Install

```bash
npm install --save yomtor
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'yomtor'
import 'yomtor/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
